738 __STATIC_INLINE void NVIC_SystemReset(void)
739 {
740  __DSB(); /* Ensure all outstanding memory accesses included
741  buffered write are completed before reset */
742  SCB->AIRCR = ((0x5FAUL << SCB_AIRCR_VECTKEY_Pos) |
743  SCB_AIRCR_SYSRESETREQ_Msk);
744  __DSB(); /* Ensure completion of memory access */
745 
746  for(;;) /* wait until reset */
747  {
748  __NOP();
749  }
