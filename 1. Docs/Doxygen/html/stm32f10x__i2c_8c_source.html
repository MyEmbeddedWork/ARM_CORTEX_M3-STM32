191 void I2C_Init(I2C_TypeDef* I2Cx, I2C_InitTypeDef* I2C_InitStruct)
192 {
193  uint16_t tmpreg = 0, freqrange = 0;
194  uint16_t result = 0x04;
195  uint32_t pclk1 = 8000000;
196  RCC_ClocksTypeDef rcc_clocks;
197  /* Check the parameters */
198  assert_param(IS_I2C_ALL_PERIPH(I2Cx));
199  assert_param(IS_I2C_CLOCK_SPEED(I2C_InitStruct->I2C_ClockSpeed));
200  assert_param(IS_I2C_MODE(I2C_InitStruct->I2C_Mode));
201  assert_param(IS_I2C_DUTY_CYCLE(I2C_InitStruct->I2C_DutyCycle));
202  assert_param(IS_I2C_OWN_ADDRESS1(I2C_InitStruct->I2C_OwnAddress1));
203  assert_param(IS_I2C_ACK_STATE(I2C_InitStruct->I2C_Ack));
204  assert_param(IS_I2C_ACKNOWLEDGE_ADDRESS(I2C_InitStruct->I2C_AcknowledgedAddress));
205 
206 /*---------------------------- I2Cx CR2 Configuration ------------------------*/
207  /* Get the I2Cx CR2 value */
208  tmpreg = I2Cx->CR2;
209  /* Clear frequency FREQ[5:0] bits */
210  tmpreg &= CR2_FREQ_Reset;
211  /* Get pclk1 frequency value */
212  RCC_GetClocksFreq(&rcc_clocks);
213  pclk1 = rcc_clocks.PCLK1_Frequency;
214  /* Set frequency bits depending on pclk1 value */
215  freqrange = (uint16_t)(pclk1 / 1000000);
216  tmpreg |= freqrange;
217  /* Write to I2Cx CR2 */
218  I2Cx->CR2 = tmpreg;
219 
220 /*---------------------------- I2Cx CCR Configuration ------------------------*/
221  /* Disable the selected I2C peripheral to configure TRISE */
222  I2Cx->CR1 &= CR1_PE_Reset;
223  /* Reset tmpreg value */
224  /* Clear F/S, DUTY and CCR[11:0] bits */
225  tmpreg = 0;
226 
227  /* Configure speed in standard mode */
228  if (I2C_InitStruct->I2C_ClockSpeed <= 100000)
229  {
230  /* Standard mode speed calculate */
231  result = (uint16_t)(pclk1 / (I2C_InitStruct->I2C_ClockSpeed << 1));
232  /* Test if CCR value is under 0x4*/
233  if (result < 0x04)
234  {
235  /* Set minimum allowed value */
236  result = 0x04;
237  }
238  /* Set speed value for standard mode */
239  tmpreg |= result;
240 
241  // [ILG]
242  #if defined ( __GNUC__ )
243  #pragma GCC diagnostic push
244  #pragma GCC diagnostic ignored "-Wconversion"
245  #endif
246 
247  /* Set Maximum Rise Time for standard mode */
248  I2Cx->TRISE = freqrange + 1;
249 
250  // [ILG]
251  #if defined ( __GNUC__ )
252  #pragma GCC diagnostic pop
253  #endif
254 
255  }
256  /* Configure speed in fast mode */
257  else /*(I2C_InitStruct->I2C_ClockSpeed <= 400000)*/
258  {
259  if (I2C_InitStruct->I2C_DutyCycle == I2C_DutyCycle_2)
260  {
261  /* Fast mode speed calculate: Tlow/Thigh = 2 */
262  result = (uint16_t)(pclk1 / (I2C_InitStruct->I2C_ClockSpeed * 3));
263  }
264  else /*I2C_InitStruct->I2C_DutyCycle == I2C_DutyCycle_16_9*/
265  {
266  /* Fast mode speed calculate: Tlow/Thigh = 16/9 */
267  result = (uint16_t)(pclk1 / (I2C_InitStruct->I2C_ClockSpeed * 25));
268  /* Set DUTY bit */
269  result |= I2C_DutyCycle_16_9;
270  }
271 
272  /* Test if CCR value is under 0x1*/
273  if ((result & CCR_CCR_Set) == 0)
274  {
275  /* Set minimum allowed value */
276  result |= (uint16_t)0x0001;
277  }
278  /* Set speed value and set F/S bit for fast mode */
279  tmpreg |= (uint16_t)(result | CCR_FS_Set);
280  /* Set Maximum Rise Time for fast mode */
281  I2Cx->TRISE = (uint16_t)(((freqrange * (uint16_t)300) / (uint16_t)1000) + (uint16_t)1);
282  }
283 
284  /* Write to I2Cx CCR */
285  I2Cx->CCR = tmpreg;
286  /* Enable the selected I2C peripheral */
287  I2Cx->CR1 |= CR1_PE_Set;
288 
289 /*---------------------------- I2Cx CR1 Configuration ------------------------*/
290  /* Get the I2Cx CR1 value */
291  tmpreg = I2Cx->CR1;
292  /* Clear ACK, SMBTYPE and SMBUS bits */
293  tmpreg &= CR1_CLEAR_Mask;
294  /* Configure I2Cx: mode and acknowledgement */
295  /* Set SMBTYPE and SMBUS bits according to I2C_Mode value */
296  /* Set ACK bit according to I2C_Ack value */
297  tmpreg |= (uint16_t)((uint32_t)I2C_InitStruct->I2C_Mode | I2C_InitStruct->I2C_Ack);
298  /* Write to I2Cx CR1 */
299  I2Cx->CR1 = tmpreg;
300 
301 /*---------------------------- I2Cx OAR1 Configuration -----------------------*/
302  /* Set I2Cx Own Address1 and acknowledged address */
303  I2Cx->OAR1 = (I2C_InitStruct->I2C_AcknowledgedAddress | I2C_InitStruct->I2C_OwnAddress1);
304 }
